iris %>% 
  mutate(ss=if_else(Sepal.Length<6,"Y","N")) %>% 
  summarise(mean=mean(Sepal.Length),sd=sd(Sepal.Length),.by=c(Species,ss)) %>% 
  ggplot(aes(x=Species,y=mean))+
  geom_col(aes(fill=ss),
         #  width=0.8, # 柱子组宽度，默认0.9
           just=0.5, # 柱子对齐位置，默认0.5 居中刻度对齐
           position=position_dodge2(preserve = "single",
                                 #   padding=0.1 # 同组柱子间距
                                    ))+
  geom_errorbar(aes(ymin=mean,ymax=mean+sd),
          #      width=0.8,
                position=position_dodge2(preserve = "single",
                                         padding=0.5
                                  ))+
  geom_text(aes(y=mean+sd,label=round(mean+sd,2),
                 hjust=0.5, # 左对齐
                 vjust=-0.5), # 下对齐
             size=5,
             fontface=3, # 字体样式
             position=position_dodge2(
               width=0.9,
               preserve = "single",
               padding=0.5
               ))
