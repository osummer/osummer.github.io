Duncan_Test

#' Duncan_Test
#' @param data dataset
#' @param xx x-variable
#' @param yy y-variable
#'
#' @return data frame
#' @export
#'
#' @import dplyr
#' @import tibble
#' @import ggplot2
#' @import tidyr
#' @import patchwork
#' @examples
#' Duncan_Test(iris,Species,Sepal.Length)

Duncan_Test<-function(data,xx,yy){
  trt<- as.character(substitute(xx))

  aov1<-aov(formula(substitute(yy~xx)),data)

  res <- aov1 %>% agricolae::duncan.test(trt=trt)

  res_mean <- res$means %>% as_tibble(rownames=trt) %>% select(1:4)

  res_group <- res$groups %>% as_tibble(rownames=trt)%>% select(1,3)

  multicom_result <- res_mean %>% left_join(res_group,by=trt)

  result<-list(anova_test=summary(aov1), multi_test=multicom_result)

  return(result)
}

多变量多重比较

长表输出sss_longer

#' sss_longer
#' @param data dataset
#' @param xx xx-variable
#' @param yy1 starts y-variable
#' @param yy2 ends y-variable
#'
#' @return data frame
#' @export
#'
#' @examples
#' iris %>% sss_longer(Species,Sepal.Length,Petal.Width)

sss_longer <- function(p,xx,yy1,yy2){
 res <- p %>%
    pivot_longer(cols={{yy1}}:{{yy2}},names_to = "para",values_to="values") %>%
    rename(Factor={{xx}}) %>%
    group_by(para) %>%
    group_modify(~ LSD_Test(data=.x, xx=Factor,yy= values)[[2]]) %>%
    mutate(SE=std/sqrt(r),CV=std/values)

 return(res)

 }

宽表输出sss_wider

#' sss_wider
#' @param p dataset
#'
#' @return data frame
#' @export
#'
#' @examples
#' iris %>%
#' sss_longer(Species,Sepal.Length,Petal.Width) %>%
#' sss_wider()
#'

sss_wider<-function(p){
  p %>%
    mutate(ss=stringr::str_c(round(values,2),"±",round(SE,2),groups)) %>%
    select(para,Factor,ss) %>%
    pivot_wider(names_from = para,values_from =ss )
}

#' bar_plot
#' @param ppp dataset
#'
#' @return list
#' @export
#'
#' @import patchwork
#' @examples
#' iris %>%
#' sss_longer(Species,Sepal.Length,Petal.Width) %>%
#' bar_plot(ppp)

bar_plot<-function(ppp){
 gp<- ppp %>% ungroup()%>%
   group_nest(para,keep=T) %>%
   mutate(gp=purrr::map(data,~ggplot(.x,aes(x=Factor,y=values))+
                   geom_col(aes(fill=Factor),width=0.5)+
                   geom_errorbar(aes(ymin=values-SE,ymax=values+SE),width=0.25)+
                   geom_text(aes(y=values+SE,label=groups),
                             nudge_y = 0.05*(max(.x$values))
                             )+
                   labs(x="",y=unique(.x$para))+
                   scale_y_continuous(expand=expansion(0,0),
                                      limits=c(0,1.1*max(.x$values+.x$SE)))+
                   theme(axis.ticks.length.x.bottom = unit(0,"cm"))
                   ))

 p1=wrap_plots(gp$gp)+plot_layout(guides="collect")
 if(length(gp$gp)>1)
   p1+plot_annotation(tag_prefix = "(",tag_suffix = ")",tag_levels="A")
 else
   p1
}

多参数（p value、effect size BF）综合输出

方差模型分析全输出aov1

#' aov1
#' @param data data_table
#' @param var_x group_var
#' @param var_y Response_var
#'
#' @return list
#' @export
#'
#' @import rlang
#' @examples
#' iris %>% aov1(Species,Sepal.Width)

aov1 <- function(data,var_x,var_y,type=3){

  data <- tibble::as_tibble(data)

  trt<- as.character(substitute(var_x))

  aov1<-aov(formula(substitute(var_y~var_x)),data) %>%
    car::Anova(type=type)

  p_value=aov1 %>% broom::tidy() %>% filter(term==trt) %>%
    select(term,p.value)

  aovBF <- data %>% statsExpressions::oneway_anova({{var_x}},{{var_y}},type="bayes")  %>%
    distinct(log_e_bf10,r2) %>% rename(r2_bayes=r2)


  eta2=aov1 %>% effectsize::eta_squared(partial=F) %>%tibble()
  omega2=aov1 %>% effectsize::omega_squared(partial=F) %>% tibble()
  epsilon2=aov1 %>% effectsize::epsilon_squared(partial=F) %>% tibble()
  cohens_f=aov1 %>% effectsize::cohens_f(partial=F)  %>%   tibble()

  aov_effsize <-bind_cols(p_value,aovBF[1],eta2[,2] ,omega2[2],epsilon2[2],cohens_f[2],aovBF[2])

  return(aov_effsize)
}

多重比较全输出pairwise_diff

#' pairwise_diff
#' @param data data_table
#' @param var_x group_var
#' @param var_y Response_var
#'
#' @return list
#' @export
#'
#' @import rlang
#' @examples
#' iris %>% pairwise_diff(Species,Sepal.Length)

pairwise_diff <- function(data,var_x,var_y){
  data <- tibble::as_tibble(data)
  trt<- as.character(substitute(var_x))
  aov1<-aov(formula(substitute(var_y~var_x)),data)

  pairwise_aov1 <- aov1 %>% emmeans::emmeans(specs=trt) %>%
    pairs(adjust="none") %>%
    broom::tidy() %>%
    select(contrast,estimate,p.value)

  effsize_aov1 <- aov1 %>% emmeans::emmeans(specs=trt) %>%
    emmeans::eff_size(sigma=sigma(aov1),edf=df.residual(aov1)) %>%
    broom::tidy() %>%
    mutate(explain=effectsize::interpret_cohens_d(estimate, rules = "cohen1988")) %>%
    select(contrast,estimate,explain)

  diff_comb <- pairwise_aov1 %>%
    left_join(effsize_aov1,by=c("contrast")) %>%
    rename(diff=estimate.x,p=p.value,cohen_d=estimate.y) %>%
    tidyr::separate(contrast,into=c("group1","group2"),sep=" - ")

  diff_bayes <- data %>%
    ggstatsplot::pairwise_comparisons(
      x ={{var_x}}, y = {{var_y}},
      type = "bayes", bf.prior = 0.77 ) %>%
    select(group1,group2,log_e_bf10,pd)

  res <- diff_comb %>% left_join(diff_bayes,by=c("group1","group2"))

  return(res)
}

#' mmm_aov used for more response variables
#' @param data dataset
#' @param xx xx-variable
#' @param yy1 starts y-variable
#' @param yy2 ends y-variable
#'
#' @return data frame
#' @export
#'
#' @examples
#' iris %>% mmm_aov(Species,Sepal.Length,Petal.Width)

mmm_aov <- function(p,xx,yy1,yy2){

  res <- p %>%
    pivot_longer(cols={{yy1}}:{{yy2}},names_to = "para",values_to="values") %>%
    rename(Factor={{xx}}) %>%
    group_by(para) %>%
    group_modify(~ aov1(data=.x,Factor,values))
  return(res)
}

SCI 绘图主题

#' theme_sci
#' @param null
#'
#' @return list
#' @export
#'
#' @examples
#' iris %>% ggplot(aes(x=Sepal.Length,y=Sepal.Width))+
#'  geom_point(aes(color=Species))+
#'  theme_sci()
theme_sci<-function(){
  theme(
    # 作图区底色
    plot.background = element_rect(fill="NA",color="NA"),
    # 图形内颜色
    panel.background = element_rect(fill="white",colour = "black",size = 1.1),
    # 去除网格线
    panel.grid = element_blank(),
    # 刻度线
    axis.ticks.length = unit(-0.2,"cm"), #刻度线粗细
    axis.ticks=element_line(size=1), #刻度线长度
    # 刻度线标签
    axis.text=element_text(size=12,color="black"), #刻度文本大小
    axis.text.x = element_text(margin = unit(c(0.4, 0.5, 0.2, 0.5), 'cm')),
    axis.text.y = element_text(margin = unit(c(0.5, 0.4, 0.5, 0.3), 'cm')),
    # 坐标轴线
    axis.line=element_line(color=NA),
    # 坐标轴标签
    axis.title = element_text(size=14),
    axis.title.x =element_text(margin = unit(c(0.2, 0, 0.5, 0), 'cm')),
    # 图例
    legend.key = element_rect(fill=NA,colour = NA),
    legend.text=element_text(size=12),
    legend.title=element_text(size=13),
    legend.position = c(0.99,0.99),
    legend.justification = c(0.99,0.99),
    strip.text=element_text(size=20))
}
